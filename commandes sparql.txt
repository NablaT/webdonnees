Groupes associés au hardrock dont la date de création est inférieur a la date de creation du groupe initial (de référence)

SELECT DISTINCT ?list ?value ?year2
	WHERE {
		{SELECT DISTINCT ?value ?year
			WHERE {
				<http://fr.dbpedia.org/resource/Aerosmith> dbpedia-owl:genre ?value
				. <http://fr.dbpedia.org/resource/Aerosmith> dbpedia-owl:activeYearsStartYear ?year
			}
		}
		?value dbpedia-owl:wikiPageWikiLink ?list
		. ?list dbpedia-owl:activeYearsStartYear ?year2
		.filter(exists{?list dbpedia-owl:bandMember ?members} ||  exists{?list dbpedia-owl:profession ?profession})
		.filter(?list!=?value)
		.filter(?year2<?year)
	}
	ORDER BY ?year2

Requete pour recuperer tous les albums d'un chanteur


SELECT DISTINCT ?album ?date 
	WHERE {
		<http://fr.dbpedia.org/resource/Maître_Gims> dbpedia-owl:wikiPageWikiLink ?album
		. ?album dbpedia-owl:releaseDate ?date
		.filter(exists{?album rdf:type dbpedia-owl:Album})
	}
ORDER BY ?date

Requete en fonction de la chanson: On recupere toutes les chansons du meme album


SELECT DISTINCT ?list ?value 
	WHERE {
		{SELECT DISTINCT ?value 
			WHERE {
				<http://fr.dbpedia.org/resource/Yesterday> dbpedia-owl:album ?value
			}
		}
		?value dbpedia-owl:wikiPageWikiLink ?list
		.filter(exists{?list dbpedia-owl:recordDate ?members})
		.filter(?list!=?value)
	}
